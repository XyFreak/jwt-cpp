371  inline std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw,
372  std::error_code& ec) {
373  ec.clear();
374 #if OPENSSL_VERSION_NUMBER <= 0x10100003L
375  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
376  BIO_new_mem_buf(const_cast<char*>(certstr.data()), static_cast<int>(certstr.size())), BIO_free_all);
377 #else

379  BIO_new_mem_buf(certstr.data(), static_cast<int>(certstr.size())), BIO_free_all);
380 #endif

382  if (!certbio || !keybio) {

385  }
